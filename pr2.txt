    is_list(Path),
    nonvar(Value).

subterm_with_address(Term, Address, Subterm) :-
    subterm_with_address_(Address, Term, Subterm).

subterm_with_address_([], Term, Term).
subterm_with_address_([I|Rest], Term, Subterm) :-
    nth1(I, Term, Child),
    subterm_with_address_(Rest, Child, Subterm).

test_case(sum_to_n,
    input([
        clause(sum_to_n(0,0), []),
        clause(sum_to_n(N,S), [
            N > 0,
            N1 is N-1,
            sum_to_n(N1,S1),
            S is S1+N
        ])
    ]),
    expected_output([
        clause(sum_to_n(N,S), [
            S is N*(N+1)//2
        ])
    ])
).
